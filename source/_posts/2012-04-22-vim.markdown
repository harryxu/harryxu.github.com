#编译Vim

    hg clone https://vim.googlecode.com/hg/ ~/vim 
    cd ~/vim
    ./configure --with-features=huge --enable-luainterp \
                --enable-perlinterp --enable-pythoninterp \
                --enable-rubyinterp --enable-gui=gtk2 \
                --disable-largefile
    make
    sudo make install
